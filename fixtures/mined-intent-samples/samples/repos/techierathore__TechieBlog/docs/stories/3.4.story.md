### FluentAutocomplete Pattern for Tags

```razor
@inject TagSvc TagService

<FluentAutocomplete TOption="BlogTag"
                    Label="Tags"
                    Placeholder="Type to search tags..."
                    OnOptionsSearch="@OnTagSearch"
                    OptionText="@(t => t.TagName)"
                    MaximumSelectedOptions="10"
                    @bind-SelectedOptions="selectedTags" />

@code {
    private IEnumerable<BlogTag> selectedTags = new List<BlogTag>();

    private async Task OnTagSearch(OptionsSearchEventArgs<BlogTag> e)
    {
        if (string.IsNullOrWhiteSpace(e.Text))
        {
            e.Items = TagService.GetAllTags().Take(10);
        }
        else
        {
            e.Items = TagService.SearchTags(e.Text);
        }
    }
}
```

### Tag Cloud Sizing Pattern

```razor
@foreach (var tag in tags)
{
    var fontSize = CalculateFontSize(tag.PostCount, minCount, maxCount);
    <a href="/tag/@tag.Slug" class="tag" style="font-size: @(fontSize)rem;">
        @tag.TagName (@tag.PostCount)
    </a>
}

@code {
    private double CalculateFontSize(int count, int min, int max)
    {
        if (max == min) return 1.0;
        var ratio = (double)(count - min) / (max - min);
        return 0.8 + (ratio * 0.8); // Range: 0.8rem to 1.6rem
    }
}
```
